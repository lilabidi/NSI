PyScript Example

Enter a number to calculate its square:

from js import document def calculate_square(event): input_element = document.getElementById("inputNumber") result_element = document.getElementById("result") input_value = input_element.value if input_value: try: number = float(input_value) square = number ** 2 result_element.innerHTML = f"The square of {number} is {square}." except ValueError: result_element.innerHTML = "Please enter a valid number." else: result_element.innerHTML = "Input cannot be empty." # Attach the Python function to the button click event document.getElementById("calculateButton").addEventListener("click", calculate_square)
