Calculate Square

from js import document from pyscript import create_proxy def calculate_square(event): input_value = document.getElementById("inputNumber").value if input_value: number = float(input_value) square = number ** 2 document.getElementById("result").innerHTML = f"Square: {square}" else: document.getElementById("result").innerHTML = "Please enter a number." # Use create_proxy to maintain the function reference proxy_function = create_proxy(calculate_square) # Attach the proxy function to the button's click event document.getElementById("calculateButton").addEventListener("click", proxy_function)
